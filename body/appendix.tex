\zchapter{常用检查命令示例}
\begin{quote}
\begin{verbatim}
# gzip 完整性检查（仅对 *.gz）
gzip -t *.fastq.gz

# 生成或核对 MD5
md5sum *.fastq.gz > md5sum.txt
md5sum -c *.md5

# FASTQ 结构与长度一致性检查（示例）
seqkit fqchk *.fastq.gz
\end{verbatim}
\end{quote}
